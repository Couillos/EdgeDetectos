# 1. Download candles with ccxt
import ccxt, pandas as pd, numpy as np
from scipy import stats

exchange = ccxt.binance()
all_ohlcv = []
since = exchange.parse8601('2020-01-01T00:00:00Z')

while since < exchange.milliseconds():
    batch = exchange.fetch_ohlcv('BTC/USDT', '1h', since, limit=1000)
    if not batch:
        break
    all_ohlcv.extend(batch)
    since = batch[-1][0] + 1  # +1ms to avoid duplicates

df = pd.DataFrame(all_ohlcv, columns=['timestamp','open','high','low','close','volume'])
df['datetime'] = pd.to_datetime(df['timestamp'], unit='ms')
df.set_index('datetime', inplace=True)
df.drop(columns=['timestamp'], inplace=True)
df = df[~df.index.duplicated(keep='first')]


# 2. Forward returns at multiple horizons
for h in [1, 4, 6, 12, 24, 48, 72, 168]:
    df[f'fwd_{h}h'] = (df['close'].shift(-h) - df['close']) / df['close'] * 100


# 3. Define signal (example: drop > 2% in 6h)
df['signal'] = (df['close'] < df['close'].shift(6) * 0.98).astype(int)


# 4. Extract signal and all-entry returns
signal_rets = df.loc[df['signal']==1, 'fwd_24h'].dropna()
all_rets = df['fwd_24h'].iloc[:-24].dropna()  # exclude last 24 rows (no forward data)

3. STATISTICAL TESTS

--- 3a. T-TEST (is mean significantly different from 0?) ---
from scipy import stats

t_stat, t_p = stats.ttest_1samp(signal_rets, 0)
# t_p < 0.05 => mean is significantly > 0 (or < 0)
# t_p < 0.001 => highly significant


--- 3b. BOOTSTRAP CONFIDENCE INTERVALS ---
def bootstrap_ci(data, n_boot=1000):
    """Resample with replacement, compute mean each time, return 95% CI"""
    boot_means = []
    for _ in range(n_boot):
        sample = np.random.choice(data, size=len(data), replace=True)
        boot_means.append(np.mean(sample))
    return [np.percentile(boot_means, 2.5), np.percentile(boot_means, 97.5)]

ci = bootstrap_ci(signal_rets.values)
# If ci[0] > 0: edge is robust (even worst-case resampled mean is positive)


--- 3c. MONTE CARLO PERMUTATION TEST ---
np.random.seed(42)
n = len(signal_rets)  # same number of entries as the signal

mc_means = []
for _ in range(1000):  # 1000 iterations (use 5000-10000 for more precision)
    # Pick N random entries from ALL possible entries (not just signal entries)
    random_sample = np.random.choice(all_rets, size=n, replace=False)
    mc_means.append(np.mean(random_sample))

mc_p = (np.array(mc_means) >= signal_rets.mean()).mean()
# mc_p < 0.05 => signal mean beats 95%+ of random entry combinations
# mc_p = 0.001 => signal is in the top 0.1% — very strong evidence


--- 3d. KOLMOGOROV-SMIRNOV TEST ---
ks_stat, ks_p = stats.ks_2samp(signal_rets, all_rets)
# Tests if the TWO DISTRIBUTIONS are different (not just the means)
# ks_p < 0.05 => signal captures a genuinely different distribution of returns
# A significant KS with non-significant t-test means the distributions differ
#   in shape (variance, skewness, tails) even if means are similar

--- 4a. ROLLING WINDOW ANALYSIS (most important for "does edge still exist?") ---
# Window: 5040 candles (7 months on 1h), Step: 720 (1 month)

rolling_results = []
for start in range(0, len(df) - 5040, 720):
    sub = df.iloc[start:start+5040]
    sr = sub.loc[sub['signal']==1, 'fwd_24h'].dropna()
    if len(sr) > 20:
        rolling_results.append({
            'date': sub.index[-1],
            'mean': sr.mean(),
            'winrate': (sr > 0).mean() * 100,
            'count': len(sr),
            'sharpe': sr.mean() / sr.std() if sr.std() > 0 else 0
        })

rolling_df = pd.DataFrame(rolling_results)
# Key metrics:
#   - % of windows with positive mean (should be > 65%)
#   - Is the rolling mean declining over time? (edge decay)
#   - Plot as bar chart with green/red bars


--- 4b. YEAR-BY-YEAR SIGNIFICANCE ---
# T-test each year separately — edge must survive in most years

df['year'] = df.index.year
yearly = []
for year, grp in df.groupby('year'):
    sr = grp.loc[grp['signal']==1, 'fwd_24h'].dropna()
    if len(sr) > 5:
        t_stat, t_p = stats.ttest_1samp(sr, 0)
        yearly.append({
            'year': year,
            'n': len(sr),
            'mean': sr.mean(),
            'winrate': (sr > 0).mean() * 100,
            'p_value': t_p,
            'sharpe': sr.mean() / sr.std() if sr.std() > 0 else 0
        })

yearly_df = pd.DataFrame(yearly)
# Count how many years have p < 0.05


--- 4c. VOLATILITY REGIME ANALYSIS ---
# Check if edge works differently in high/low volatility

df['realized_vol_24h'] = df['close'].pct_change().rolling(24).std() * np.sqrt(24) * 100
df['vol_regime'] = pd.qcut(df['realized_vol_24h'], 3, labels=['Low', 'Mid', 'High'])

regime_results = []
for regime, grp in df.groupby('vol_regime'):
    sr = grp.loc[grp['signal']==1, 'fwd_24h'].dropna()
    ar = grp['fwd_24h'].dropna()  # all entries in this regime
    if len(sr) > 10:
        t_stat, t_p = stats.ttest_1samp(sr, 0)
        excess = sr.mean() - ar.mean()  # excess return vs random in same regime
        regime_results.append({
            'regime': regime,
            'n': len(sr),
            'mean': sr.mean(),
            'excess_vs_random': excess,
            'winrate': (sr > 0).mean() * 100,
            'p_value': t_p
        })

10. KEY NOTES
===============================================================================

- All forward returns use CLOSE prices (not high/low). In real trading you
  would enter at market (close) and exit at market (close T+H).

- Transaction costs are NOT included. Binance maker fee ~0.1%, taker ~0.1%.
  Round trip = ~0.2%. So for 1h horizon (+0.02% edge), the edge is dead
  after costs. For 24h (+0.26-0.44%), it survives but is reduced.

- Slippage is NOT included. In volatile conditions (which these signals
  often trigger in), slippage can be 0.05-0.2% per side.

- The Monte Carlo p-value answers: "If I entered at N random times,
  what fraction would have a higher mean return than my signal?"
  Lower = better. p=0.001 means top 0.1%.

- The KS test answers: "Are the return DISTRIBUTIONS fundamentally
  different?" Even if means are similar, different shapes indicate
  the signal captures unique information.

- Bootstrap CI answers: "Given the observed sample, what's the range
  of plausible true mean values?" If the 95% CI excludes 0, the
  edge is robust to sampling variability.

- Rolling window analysis is THE key test for edge persistence.
  An edge that worked in 2021 but not 2024-2026 is not reliable.
  Look for >70% of windows positive and no clear downward trend.

- np.random.seed(42) is set before each Monte Carlo for reproducibility.

- The "all returns" baseline (for KS and MC) excludes the last H candles
  because those don't have forward data: df['fwd_24h'].iloc[:-24]
